#!/usr/bin/env node
/**
 * k6 Summary JSON Merger
 *
 * Usage:
 *   node merge-k6-summaries.js file1.json file2.json ... [--output merged.json] [--mode sequential|parallel]
 *
 * --mode sequential  (default) Runs one after another вҶ’ VUs = max
 * --mode parallel    Runs at same time               вҶ’ VUs = sum
 */

const fs = require("fs");

// в”Җв”Җв”Җ CLI в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ

const rawArgs = process.argv.slice(2);
let mode = "sequential";
let outputFile = "merged-k6-summary.json";
const inputFiles = [];

for (let i = 0; i < rawArgs.length; i++) {
  if (rawArgs[i] === "--output") { outputFile = rawArgs[++i]; }
  else if (rawArgs[i] === "--mode") {
    const m = rawArgs[++i];
    if (!["sequential", "parallel"].includes(m)) {
      console.error(`вқҢ Invalid mode "${m}". Use: sequential | parallel`); process.exit(1);
    }
    mode = m;
  } else { inputFiles.push(rawArgs[i]); }
}

if (inputFiles.length < 1) {
  console.error("Usage: node merge-k6-summaries.js file1.json file2.json [--output out.json] [--mode sequential|parallel]");
  process.exit(1);
}
for (const f of inputFiles) {
  if (!fs.existsSync(f)) { console.error(`вқҢ File not found: ${f}`); process.exit(1); }
}

// в”Җв”Җв”Җ Helpers в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ

const VUS_METRICS     = new Set(["vus", "vus_max"]);
const COUNTER_METRICS = new Set(["http_reqs", "iterations", "data_received", "data_sent"]);
const RATIO_METRICS   = new Set(["checks", "http_req_failed"]);

function weightedAvg(instances, field, weights) {
  const total = weights.reduce((a, b) => a + b, 0);
  if (total === 0) return 0;
  return instances.reduce((acc, m, i) => acc + (m[field] ?? 0) * weights[i], 0) / total;
}

function mergeMetric(key, instances, reqCounts, iterCounts, mode) {
  const result = {};
  const isVus    = VUS_METRICS.has(key);
  const isRatio  = RATIO_METRICS.has(key);
  const isTiming = !isVus && !COUNTER_METRICS.has(key) && !isRatio;
  const weights  = isTiming ? reqCounts : iterCounts;

  if ("min"    in instances[0]) result.min      = Math.min(...instances.map(m => m.min));
  if ("max"    in instances[0]) result.max      = Math.max(...instances.map(m => m.max));
  if ("avg"    in instances[0]) result.avg      = weightedAvg(instances, "avg",    weights);
  if ("med"    in instances[0]) result.med      = weightedAvg(instances, "med",    weights);
  if ("p(90)"  in instances[0]) result["p(90)"] = weightedAvg(instances, "p(90)", weights);
  if ("p(95)"  in instances[0]) result["p(95)"] = weightedAvg(instances, "p(95)", weights);
  if ("count"  in instances[0]) result.count    = instances.reduce((s, m) => s + (m.count  ?? 0), 0);
  if ("rate"   in instances[0]) result.rate     = instances.reduce((s, m) => s + (m.rate   ?? 0), 0);
  if ("passes" in instances[0]) result.passes   = instances.reduce((s, m) => s + (m.passes ?? 0), 0);
  if ("fails"  in instances[0]) result.fails    = instances.reduce((s, m) => s + (m.fails  ?? 0), 0);

  if ("value" in instances[0]) {
    if (isRatio) {
      const total = (result.passes ?? 0) + (result.fails ?? 0);
      result.value = total > 0 ? result.passes / total : 0;
    } else if (isVus) {
      result.value = mode === "parallel"
        ? instances.reduce((s, m) => s + (m.value ?? 0), 0)
        : Math.max(...instances.map(m => m.value ?? 0));
    } else {
      result.value = instances.reduce((s, m) => s + (m.value ?? 0), 0);
    }
  }
  return result;
}

function mergeChecks(checksArray) {
  const merged = {};
  const allNames = new Set(checksArray.flatMap(c => Object.keys(c)));
  for (const name of allNames) {
    const instances = checksArray.map(c => c[name]).filter(Boolean);
    merged[name] = {
      name:   instances[0].name,
      path:   instances[0].path,
      id:     instances[0].id,
      passes: instances.reduce((s, x) => s + (x.passes ?? 0), 0),
      fails:  instances.reduce((s, x) => s + (x.fails  ?? 0), 0),
    };
  }
  return merged;
}

// в”Җв”Җв”Җ Merge в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ

const summaries  = inputFiles.map(f => JSON.parse(fs.readFileSync(f, "utf8")));
const reqCounts  = summaries.map(s => s.metrics?.http_reqs?.count  || 1);
const iterCounts = summaries.map(s => s.metrics?.iterations?.count || 1);

const allMetricKeys = new Set(summaries.flatMap(s => Object.keys(s.metrics || {})));
const mergedMetrics = {};
for (const key of allMetricKeys) {
  const instances = summaries.map(s => s.metrics?.[key]).filter(Boolean);
  if (instances.length > 0) mergedMetrics[key] = mergeMetric(key, instances, reqCounts, iterCounts, mode);
}

const rootGroups = summaries.map(s => s.root_group).filter(Boolean);
const firstGroup = rootGroups[0] || {};
const mergedRootGroup = {
  name:   firstGroup.name   ?? "",
  path:   firstGroup.path   ?? "",
  id:     firstGroup.id     ?? "",
  groups: firstGroup.groups ?? {},
  checks: mergeChecks(rootGroups.map(g => g.checks || {})),
};

// в”Җв”Җ Output JSON вҖ” same shape as input files в”Җв”Җ
const merged = {
  root_group: mergedRootGroup,
  metrics: mergedMetrics,
};

fs.writeFileSync(outputFile, JSON.stringify(merged, null, 4), "utf8");

// в”Җв”Җв”Җ Console Analysis Table в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ

const m = merged.metrics;
const checks = mergedRootGroup.checks;
const W = 68;
const line  = "в”Җ".repeat(W);
const dline = "в•җ".repeat(W);

function pad(str, len, right = false) {
  str = String(str);
  if (right) return str.padStart(len);
  return str.padEnd(len);
}

function row(label, value, note = "") {
  const l = pad(label, 36);
  const v = pad(value, 18, true);
  const n = note ? `   ${note}` : "";
  console.log(`в”Ӯ  ${l}${v}  в”Ӯ${n}`);
}

function sectionHeader(title) {
  console.log(`в”ң${line}в”Ө`);
  const spaces = W - 2 - title.length;
  const left = Math.floor(spaces / 2);
  const right = spaces - left;
  console.log(`в”Ӯ ${" ".repeat(left)}${title}${" ".repeat(right)} в”Ӯ`);
  console.log(`в”ң${"в”Җ".repeat(W)}в”Ө`);
}

function blank() {
  console.log(`в”Ӯ${" ".repeat(W)}в”Ӯ`);
}

// Header
console.log(`\nв•”${dline}в•—`);
const hdr = "k6 MERGED PERFORMANCE REPORT";
const hp = Math.floor((W - hdr.length) / 2);
console.log(`в•‘${" ".repeat(hp)}${hdr}${" ".repeat(W - hp - hdr.length)}в•‘`);
console.log(`в•ҡ${dline}в•қ`);

console.log(`в”Ң${line}в”җ`);
row("Mode",         mode === "parallel" ? "PARALLEL (VUs summed)" : "SEQUENTIAL (VUs max'd)");
row("Files merged", inputFiles.length);
row("Output",       outputFile);
blank();

sectionHeader("TRAFFIC SUMMARY");
row("Total HTTP Requests",  m.http_reqs?.count?.toLocaleString() ?? "N/A");
row("Request Rate",         `${(m.http_reqs?.rate ?? 0).toFixed(2)} req/s`);
row("Total Iterations",     m.iterations?.count?.toLocaleString() ?? "N/A");
row("Iteration Rate",       `${(m.iterations?.rate ?? 0).toFixed(2)} iter/s`);
row("Data Received",        `${((m.data_received?.count||0)/1024).toFixed(1)} KB  @  ${((m.data_received?.rate||0)/1024).toFixed(1)} KB/s`);
row("Data Sent",            `${((m.data_sent?.count||0)/1024).toFixed(1)} KB  @  ${((m.data_sent?.rate||0)/1024).toFixed(1)} KB/s`);
blank();

sectionHeader("HTTP REQUEST DURATION (ms)");
row("Average",  (m.http_req_duration?.avg ?? 0).toFixed(3));
row("Median",   (m.http_req_duration?.med ?? 0).toFixed(3),          "вҡ  approx");
row("Min",      (m.http_req_duration?.min ?? 0).toFixed(3));
row("Max",      (m.http_req_duration?.max ?? 0).toFixed(3));
row("p(90)",    (m.http_req_duration?.["p(90)"] ?? 0).toFixed(3),    "вҡ  approx");
row("p(95)",    (m.http_req_duration?.["p(95)"] ?? 0).toFixed(3),    "вҡ  approx");
blank();

sectionHeader("REQUEST PHASE BREAKDOWN (avg ms)");
row("Blocked",         (m.http_req_blocked?.avg         ?? 0).toFixed(3));
row("Connecting",      (m.http_req_connecting?.avg      ?? 0).toFixed(3));
row("TLS Handshaking", (m.http_req_tls_handshaking?.avg ?? 0).toFixed(3));
row("Sending",         (m.http_req_sending?.avg         ?? 0).toFixed(3));
row("Waiting (TTFB)",  (m.http_req_waiting?.avg         ?? 0).toFixed(3));
row("Receiving",       (m.http_req_receiving?.avg       ?? 0).toFixed(3));
blank();

sectionHeader("CHECKS");
for (const [, c] of Object.entries(checks)) {
  const total = c.passes + c.fails;
  const pct   = total > 0 ? ((c.passes / total) * 100).toFixed(1) : "0.0";
  const icon  = c.fails === 0 ? "вң”" : "вңҳ";
  row(`${icon}  ${c.name}`, `${c.passes} / ${total}`, `${pct}%`);
}
row("Total passes",  m.checks?.passes ?? 0);
row("Total fails",   m.checks?.fails  ?? 0);
blank();

sectionHeader("VIRTUAL USERS");
row(`Active VUs  (${mode === "parallel" ? "summed" : "max'd"})`, m.vus?.value     ?? "N/A");

blank();

console.log(`в””${line}в”ҳ`);

console.log("  вҡ    med / p(90) / p(95) are weighted approximations\n");

const m      = mergedMetrics;
const checks = mergedRootGroup.checks;

// Collect per-API entries from tagged metrics e.g. http_req_duration{name:GET /api/foo}
const apiMap = {};
for (const [key, val] of Object.entries(m)) {
  const match = key.match(/\{name:([^,}]+)/);
  if (!match) continue;
  const name = match[1].trim();
  if (!apiMap[name]) apiMap[name] = {};
  if (key.startsWith("http_req_duration")) apiMap[name].duration = val;
}

// в”Җв”Җ API Summary Table в”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җв”Җ

const totalReqs   = m.http_reqs?.count  ?? 0;
const totalPasses = m.checks?.passes    ?? 0;
const iters       = m.iterations?.count ?? 0;
const apiNames    = Object.keys(apiMap);

const apiRows = apiNames.map(name => {
  const d = apiMap[name].duration ?? {};
  return [name, iters, iters, fmt(d.avg), fmt(d["p(90)"]), fmt(d["p(95)"])];
});

// OVERALL always appended as last row
apiRows.push([
  "OVERALL",
  totalReqs.toLocaleString(),
  totalPasses.toLocaleString(),
  fmt(dur.avg),
  fmt(dur["p(90)"]),

]);
